has_covenant_with_bts_patron = {
	OR = {
		# Not present in Utopia DLC
		# Shroud DLC check
		has_covenant = bts_patron
	}
}

bts_num_minor_patron_relations = {
	check_variable_arithmetic = {
		which = value:num_minor_patron_relations
		value >= $COUNT$
	}
}

bts_num_minor_patron_relations_with_tooltip = {
	custom_tooltip = {
		bts_num_minor_patron_relations = { COUNT = $COUNT$ }
		fail_text = bts_requires_bts_num_minor_patron_$COUNT$
	}
}


bts_has_any_covenant = {
	OR = {
		has_any_covenant = yes
		has_covenant_with_bts_patron = yes
	}
}

bts_has_zroni_shield = {
    OR = {
        has_component = bts_small_zroni_shield
        has_component = bts_medium_zroni_shield
        has_component = bts_large_zroni_shield
    }
}
