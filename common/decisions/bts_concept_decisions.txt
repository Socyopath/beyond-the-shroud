decision_bts_concept_time = {
	owned_planets_only = no
	sound = event_energy_field
	icon = decision_wormhole

	resources = {
		category = decisions
		cost = {

		}
	}

	potential = {
		is_planet_class = pc_bts_labyrinth_of_abstraction
		exists = owner
		owner = {
			has_tradition = tr_concept_time_adopt
		}
	}

    allow = {
		owner = {
			custom_tooltip = {
				fail_text = "bts_decision_concept_already_use"
				NOT = {
					any_owned_ship = {
						has_ship_flag = bts_concept_time
					}
				}
			}
			custom_tooltip = {
				fail_text = "bts_decision_concept_cooldown"
				NOT = {
					has_country_flag = bts_decision_concept_cooldown
				}
			}
		}
    }

	effect = {
		custom_tooltip = decision_bts_concept_time_effect
		hidden_effect = {
			bts_create_concept_ship = {
				NAME = decision_bts_concept_time
				FLAG = bts_concept_time
			}
		}
	}
}
