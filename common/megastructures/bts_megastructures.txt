					}
				}
			}

            }
        }
    }
	placement_rules = {

				}
			}

				fail_text = "requires_no_anomaly"
				has_anomaly = no
			}
			custom_tooltip = {

			}
		}

	on_build_cancel = {}

		spawn_planet = {

			orbit_distance_offset = 10
			has_ring = no

                remove_all_buildings = yes
                add_building = bts_labyrinth_of_abstraction_core_1
			}
		}
		remove_megastructure = fromfrom
	}
}
