bts_patron_aura_max_intensity_effect = {
	if = {
        limit = {
            any_system_planet = { is_astral_scar = yes }
        }
        
        random_system_planet = {
            limit = { is_astral_scar = yes }
            split_astral_scar = yes
        }

        create_message = {
            type = MESSAGE_PSIONIC_AURA_whisperers_in_the_void
            localization = MESSAGE_PSIONIC_AURA_MAX_INTENSITY_bts_patron_split_DESC
            days = 30
            variable = {
                type = name
                localization = SYSTEM
                scope = this
            }
        }
    }
    else_if = {
        limit = {
            has_astral_rift = yes
        }
        random_astral_rift = {
            create_fleet = {
                name = "NAME_Dimensional_Fleet"
                settings = {
                    can_upgrade = no
                    uses_naval_capacity = no
                    spawn_debris = no
                }
                effect = {
                    set_owner = root.aura_owner
                    while = {
                        count = @bts_patron_aura_effect_dimensional_cruiser_count
                        create_ship = {
                            name = "NAME_Astral_Avatar"
                            design = "NAME_Astral_Avatar"
                            upgradable = no
                            graphical_culture = "extra_dimensional_01"
                        }
                    }
                    while = {
                        count = @bts_patron_aura_effect_dimensional_destroyer_count
                        create_ship = {
                            name = "NAME_Astral_Avatar_Small"
                            design = "NAME_Astral_Avatar_Small"
                            upgradable = no
                            graphical_culture = "extra_dimensional_01"
                        }
                    }
                    set_location = prev
                }
            }
        }
        create_message = {
            type = MESSAGE_PSIONIC_AURA_whisperers_in_the_void
            localization = MESSAGE_PSIONIC_AURA_MAX_INTENSITY_bts_patron_fleet_DESC
            days = 30
            variable = {
                type = name
                localization = SYSTEM
                scope = this
            }
        }
    }
}

bts_set_patron_attunement_exact = {
    if = {
        limit = { has_patron_relation = { patron = the_eater_of_worlds } }
        set_attunement = { patron = the_eater_of_worlds value = 0 }
    }
    if = {
        limit = { has_patron_relation = { patron = the_cradle_of_souls } }
        set_attunement = { patron = the_cradle_of_souls value = 0 }
    }
    if = {
        limit = { has_patron_relation = { patron = the_composer_of_strands } }
        set_attunement = { patron = the_composer_of_strands value = 0 }
    }
    if = {
        limit = { has_patron_relation = { patron = the_instrument_of_desire } }
        set_attunement = { patron = the_instrument_of_desire value = 0 }
    }
    if = {
        limit = { has_patron_relation = { patron = whisperers_in_the_void } }
        set_attunement = { patron = whisperers_in_the_void value = 0 }
    }
    if = {
        limit = { has_patron_relation = { patron = bts_patron } }
        set_attunement = { patron = bts_patron value = 0 }
    }
    set_attunement = { patron = $PATRON$ value = $VALUE$ }
}
