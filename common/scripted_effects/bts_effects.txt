# Ship
bts_update_ship_shield_reduction_zroni_shield = {
	if = {
		limit = {
			bts_has_zroni_shield = yes
            NOT = {	has_modifier = bts_zroni_shield_reduction_protection }
			fleet.solar_system = {
				has_modifier = zroni_storm
			}
		}
		add_modifier = { modifier = bts_zroni_shield_reduction_protection }
	}
	else_if = {
		limit = {
            has_modifier = bts_zroni_shield_reduction_protection
			fleet.solar_system = {
				NOT = {
					has_modifier = zroni_storm
				}
			}
		}
		remove_modifier = bts_zroni_shield_reduction_protection
	}
	else = {
		
	}
}
