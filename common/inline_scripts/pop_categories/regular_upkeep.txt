# Food Upkeep
upkeep = {
	trigger = {
		NOT = { owner = { has_technology = bts_tech_zro_infusion } }
		has_food_upkeep = yes
		is_phototrophic = no
	}
	food = @living_standard_food_normal
}
upkeep = {
	trigger = {
		has_food_upkeep = yes
		is_phototrophic = yes
	}
	food = @living_standard_phototrophic_normal
}

# Zro Infusion Upkeep
upkeep = {
	trigger = {
		owner = { has_technology = bts_tech_zro_infusion }
		has_food_upkeep = yes
		is_phototrophic = no
	}
	sr_zro = @bts_tech_zro_infusion_zro_upkeep_cost
}
upkeep = {
	trigger = {
		has_food_upkeep = yes
		is_phototrophic = yes
	}
	food = @living_standard_phototrophic_normal
}
