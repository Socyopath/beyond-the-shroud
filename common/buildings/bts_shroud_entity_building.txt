
# Colony
bts_shroud_entity_core_1 = {
	capital = yes
	can_build = no
	can_demolish = no
	can_be_ruined = no
	can_be_disabled = no
	position_priority = 0
	category = government

	potential = {
		is_planet_class = pc_bts_shroud_entity
	}
	upgrades = {
		bts_shroud_entity_core_2
	}

	planet_modifier = {
		planet_housing_add = 10
		job_bts_new_add = 10
	}
	resources = {
		category = planet_buildings
		upkeep = {
			energy = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
		}
		planet_max_buildings_add = -1
		mult = owner.modifier:planet_max_buildings_add
	}

	# inline_script = {
	# 	script = buildings/on_all_capital_buildings
	# }
}
